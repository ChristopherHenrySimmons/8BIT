CREATE DATABASE 8BITDB;

USE 8BITDB;

CREATE TABLE IMAGES ( 
IMAGES_ID INTEGER UNSIGNED NOT NULL UNIQUE AUTO_INCREMENT, 
IMAGES_Caption VARCHAR(45), 
IMAGES_Img LONGBLOB NOT NULL, 
PRIMARY KEY(IMAGES_ID) 
);

CREATE TABLE ADMIN (
  ADMIN_ID INT(11) NOT NULL UNIQUE AUTO_INCREMENT,
  ADMIN_Email VARCHAR(50) NOT NULL,
  ADMIN_Password VARCHAR(50) NOT NULL,
PRIMARY KEY(ADMIN_ID)
);

CREATE TABLE ACCOUNT (
  ACCOUNT_ID INT(11) NOT NULL UNIQUE AUTO_INCREMENT,
  ACCOUNT_Email VARCHAR(50) NOT NULL,
  ACCOUNT_Password VARCHAR(50) NOT NULL,
  ACCOUNT_Status int(11) NOT NULL DEFAULT 0,
PRIMARY KEY(ACCOUNT_ID)
);

CREATE TABLE PRODUCT (
  PRODUCT_ID INT(11) NOT NULL UNIQUE AUTO_INCREMENT,
  PRODUCT_Type VARCHAR(50) NOT NULL,
  PRODUCT_Title VARCHAR(50) NOT NULL,
  PRODUCT_Price INT(11) NOT NULL,
  PRODUCT_Version VARCHAR(20) NOT NULL,
PRIMARY KEY(PRODUCT_ID)
);

CREATE TABLE NEWS (
  ARTICLE_ID INT(11) NOT NULL UNIQUE AUTO_INCREMENT,
  ARTICLE_Title VARCHAR(50) NULL,
  ARTICLE_Category VARCHAR(50) NULL,
  ARTICLE_Date DATE NULL,
  ARTICLE_Message VARCHAR(50) NULL,
PRIMARY KEY(ARTICLE_ID)
);

CREATE TABLE ORDERS (
  ORDERS_ID INT(11) NOT NULL UNIQUE AUTO_INCREMENT,
  ORDERS_CreiditCard INT(20) NOT NULL UNIQUE,
  ORDERS_ExpiryDate DATE NOT NULL,
  ORDERS_SecurityNumber INT(6) NOT NULL,
  ORDERS_NameOnCard VARCHAR(200) NOT NULL,
  ORDERS_DATE DATE NOT NULL,
  ACCOUNT_ID INT,
  PRODUCT_ID INT,
PRIMARY KEY(ORDERS_ID),
FOREIGN KEY (ACCOUNT_ID) REFERENCES ACCOUNT(ACCOUNT_ID),
FOREIGN KEY (PRODUCT_ID) REFERENCES PRODUCT(PRODUCT_ID)
);
